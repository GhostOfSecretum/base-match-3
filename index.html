Deploy a simple storage smart contract on Base network with one click!

SimpleStorage.sol 
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract SimpleStorage {
    uint256 public storedValue;
    address public owner;
    
    event ValueChanged(uint256 newValue);
    
    constructor() {
        owner = msg.sender;
        storedValue = 0;
    }
    
    function setValue(uint256 _value) public {
        storedValue = _value;
        emit ValueChanged(_value);
    }
    
    function getValue() public view returns (uint256) {
        return storedValue;
    }
}

How to Play

Goal

Match 3 or more crypto tokens by swapping adjacent tiles to reach the target score before running out of moves!

How to Match

Click and drag (or tap and swipe) adjacent tiles to swap them. When 3 or more matching tokens line up horizontally or vertically, they will be removed and new tokens will fall into place.

Combos & Multipliers

Create multiple matches in quick succession to build up your combo multiplier! Each combo increases your score multiplier, making it easier to reach the target.

Special Tokens

Match 4 or 5 tokens to create special tokens with powerful effects. These can clear entire rows, columns, or large areas of the board!

Leaderboard

Connect your wallet to save your high scores and compete with other players on the global leaderboard!

GM Streak

Play at least one game every day to maintain your streak! The longer your streak, the more rewards you can unlock.
